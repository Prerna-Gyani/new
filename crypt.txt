# Substitution (Caesar) Cipher
def caesar_encrypt_simple(text, key):
    result = ""
    for c in text:
        if c.islower():
            shifted = ord(c) - ord('a')
            shifted = shifted + key
            shifted = shifted % 26
            new_char = chr(shifted + ord('a'))
            result = result + new_char
        elif c.isupper():
            shifted = ord(c) - ord('A')
            shifted = shifted + key
            shifted = shifted % 26
            new_char = chr(shifted + ord('A'))
            result = result + new_char
        else:
            result = result + c
    return result

def caesar_encrypt(text, key):
    return ''.join(
        chr((ord(c) - ord('a') + key) % 26 + ord('a')) if c.islower() else
        chr((ord(c) - ord('A') + key) % 26 + ord('A')) if c.isupper() else c
        for c in text)
def caesar_decrypt(cipher, key):
    return caesar_encrypt(cipher, -key)

# Vigenère Cipher
def vigenere_encrypt(text, key):
    key = key.lower()
    letters = [c for c in text if c.isalpha()]
    shifts = [ord(key[i % len(key)]) - ord('a') for i in range(len(letters))]
    it = iter(shifts)
    return ''.join(
        chr((ord(c) - ord('a') + next(it)) % 26 + ord('a')) if c.islower() else
        chr((ord(c) - ord('A') + next(it)) % 26 + ord('A')) if c.isupper() else c
        for c in text)
def vigenere_decrypt(cipher, key):
    key = key.lower()
    letters = [c for c in cipher if c.isalpha()]
    shifts = [ord(key[i % len(key)]) - ord('a') for i in range(len(letters))]
    it = iter(shifts)
    return ''.join(
        chr((ord(c) - ord('a') - next(it)) % 26 + ord('a')) if c.islower() else
        chr((ord(c) - ord('A') - next(it)) % 26 + ord('A')) if c.isupper() else c
        for c in cipher)

# Affine Cipher
def affine_encrypt(text, a, b):
    return ''.join(
        chr((a * (ord(c) - ord('a') ) + b) % 26 + ord('a')) if c.islower() else
        chr((a * (ord(c) - ord('A') ) + b) % 26 + ord('A')) if c.isupper() else c
        for c in text)
def modinv(a, m):
    for x in range(1, m):
        if (a * x) % m == 1:
            return x
    raise ValueError('No modular inverse')
def affine_decrypt(cipher, a, b):
    a_inv = modinv(a, 26)
    return ''.join(
        chr((a_inv * ((ord(c) - ord('a') - b)) % 26) + ord('a')) if c.islower() else
        chr((a_inv * ((ord(c) - ord('A') - b)) % 26) + ord('A')) if c.isupper() else c
        for c in cipher)
if __name__ == "__main__":

#padding when there is 16bit and to store is only 6bits then to fill remaining space is filled by #different bits is padding & emptying it is unpadding.
from Crypto.Cipher import AES
from Crypto.Random import get_random_bytes
from Crypto.Util.Padding import pad, unpad
def encrypt_file(input_file, output_file, key):  
    with open(input_file, 'rb') as f:
        data = f.read()  
    cipher = AES.new(key, AES.MODE_CBC)  
    encrypted = cipher.encrypt(pad(data, AES.block_size))  
    with open(output_file, 'wb') as f:
        f.write(cipher.iv + encrypted)  
def decrypt_file(input_file, output_file, key):
    with open(input_file, 'rb') as f:
        iv = f.read(16)  
        encrypted = f.read()  
    cipher = AES.new(key, AES.MODE_CBC, iv)  
    decrypted = unpad(cipher.decrypt(encrypted), AES.block_size) 
    with open(output_file, 'wb') as f:
        f.write(decrypted) 
def main():
    # Generate a random 16-byte (128-bit) AES key
    key = get_random_bytes(16)
    input_file = input("Enter the path of the input file to encrypt/decrypt: ").strip()
    encrypt_file(input_file, 'encrypt.txt', key)
    print(f"File encrypted at: ")
    print(encrypt_file)
    decrypt_file('encrypt.txt', 'decrypt.txt', key)
    print(f"File decrypted ")

    main()

from stegano import lsb
from PIL import Image
import os
def hide_message(image_path, message, output_path=None):
    base, ext = os.path.splitext(image_path)
    # Convert to PNG if JPG (stegano requires lossless format)
    if ext.lower() in ['.jpg', '.jpeg']:
        png_path = base + '.png'
        Image.open(image_path).save(png_path)
        image_path = png_path
    # Set output path
    if output_path is None:
        output_path = base + '-secret.png'
    steg = lsb.hide(image_path, message)
    steg.save(output_path)
    print(f'Message hidden in: {output_path}')
    return output_path
def reveal_message(image_path):
    """Reveal a hidden message from an image"""
    msg = lsb.reveal(image_path)
    return msg
if __name__ == "__main__":
    input_image = input('Enter image path (JPG/PNG): ')
    message = input('Enter secret message: ')
    output = hide_message(input_image, message)
    revealed = reveal_message(output)
    print('Hidden message:', revealed)

import hashlib

# Initial sent message
sent_msg = input('Enter message: ')

s_md_l = hmac.new(key=key, msg=sent_msg.encode(), digestmod=hashlib.sha256)
init_msg_digest = s_md_l.hexdigest()
#Received message
received = sent_msg
r_md_l = hmac.new(key=key, msg=received.encode(), digestmod=hashlib.sha256)
recv_msg_digest = r_md_l.hexdigest()
#Comparing sent and received msg
print('----Before Tampering-----')
print('Original Message:', sent_msg)
print('Original Digest:', init_msg_digest)
print('Received Message:', received)
print('Received Digest:', recv_msg_digest)
print('Is the message received without any tampering:', 'Yes' if hmac.compare_digest(init_msg_digest, recv_msg_digest) else 'No')
# Tampered message
tampered_msg = sent_msg[1:] if len(sent_msg) > 0 else sent_msg  # Avoid error if empty
md_2 = hmac.new(key=key, msg=tampered_msg.encode(), digestmod=hashlib.sha256)
tampered_msg_digest = md_2.hexdigest()
# Comparing after tampering
print('----After Tampering-----')
print('Tampered Message:', tampered_msg)
print('Tampered Digest:', tampered_msg_digest)
print('Is the message received without any tampering:', 'Yes' if hmac.compare_digest(init_msg_digest, tampered_msg_digest) else 'No')

Code: 
⇒ server.py
import socket
from Crypto.Cipher import AES
from Crypto.Util.Padding import unpad
# shared secret key (16 bytes)
key = b'1234567890abcdef'   
s = socket.socket()
s.bind(("0.0.0.0", 5000))
s.listen(1)
conn, addr = s.accept()
print("Connected:", addr)
iv = conn.recv(16)
ciphertext = conn.recv(1024)
print("\n--- Server ---")
print("IV received (hex):", iv.hex())
print("Ciphertext received (hex):", ciphertext.hex())
cipher = AES.new(key, AES.MODE_CBC, iv)
plaintext = unpad(cipher.decrypt(ciphertext), 16)
print("Decrypted message:", plaintext.decode())
conn.close()
