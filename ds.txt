Aim: Load a Dataset, Identify Missing Values, Handle Them, and Remove Outliers.


 
Method 1 : Using Median, Mode to impute missing values
Code:
import seaborn as sns
import pandas as pd
# Load the titanic dataset into a pandas DataFrame (built-in dataset)
df = sns.load_dataset('titanic')
# Display the first few rows
print(df.head())
# Identify missing values
print("Missing Values: \n", df.isnull().sum())
# Handle missing values - imputing (filling missing values)
# Fill numerical columns with the mdeian
# Fill categorical columns with the mode 
for col in df.select_dtypes(include=['number']):
    df.fillna({'col':df[col].median()}, inplace=True)
for col in df.select_dtypes(include=['object']):
    df.fillna({'col':df[col].mode()[0]}, inplace=True)
#  Detect and remove outliers using IQR method
Q1 = df.select_dtypes(include=['number']).quantile(0.25)
Q3 = df.select_dtypes(include=['number']).quantile(0.75)
IQR = Q3 - Q1
df_cleaned = df[~((df.select_dtypes(include=['number']) < (Q1 - 1.5 * IQR)) | (df.select_dtypes(include=['number']) > (Q3 + 1.5 * IQR))).any(axis=1)]
print("\nShape before outlier removal: ", df.shape)
print("\nShape after outlier removal: ",df_cleaned.shape)
print(df.isnull().sum())
print(df.info())

Code:
# knn imputer , gets neighbors and then calculates mean
import pandas as pd
import numpy as np
import seaborn as sns
from sklearn.impute import KNNImputer
from scipy.stats import zscore
import matplotlib.pyplot as plt
# Load dataset
df = sns.load_dataset("titanic")
print("Original Shape: ", df.shape)
# Percentage of missing values
missing_percent = df.isnull().mean() * 100
print("\nMissing Percentage:\n",missing_percent)
# KNN Imputation
num_cols = df.select_dtypes(include=['number']).columns
imputer = KNNImputer(n_neighbors=5) # nearest neighbor in knn
df[num_cols] = imputer.fit_transform(df[num_cols])
# Z - Score Outlier Removal
z_scores = np.abs(zscore(df[num_cols]))
df_clean = df[(z_scores < 3 ).all(axis=1)]
print("\nShape after cleaning: ",df_clean.shape)
# Boxplot Visualization
for col in num_cols:
    sns.boxplot(x=df_clean[col])
    plt.title(f"Boxplot of {col}")
    plt.show()

Practical No. 10
Aim: Apriori Algorithm for Market Basket Analysis.



Method 1 : Code:
import pandas as pd
import seaborn as sns
from itertools import combinations
from collections import defaultdict
# Load inbuilt dataset
df = sns.load_dataset('titanic')
# Select categorical columns -  We'll treat 'passenger class', 'sex', and 'embarked' as items "in a basket" (like items bought together)
df = df[['pclass', 'sex', 'embarked']].dropna()
# Convert each row into a transaction (basket) (basket of categorical attributes) transactions 
transactions = df.astype(str).values.tolist()
# Step 1: Create frequent itemsets and count
def get_frequent_itemsets(transactions, min_support=0.1):
    item_counts = defaultdict(int)
    transaction_list = list(map(set, transactions))
    total_tx = len(transaction_list)
    freq_itemsets = []
    # Generate all combinations (size 1 to 3)
    for size in range(1, 4):
        for transaction in transaction_list:
            for itemset in combinations(transaction, size):
                item_counts[itemset] += 1
    # Filter by support (based on min support)
    for itemset, count in item_counts.items():
        support = count / total_tx
        if support >= min_support:
            freq_itemsets.append((itemset, support))
    return freq_itemsets
# Step 2: Generate association rules
def generate_rules(freq_itemsets, min_confidence=0.6):
    rules = []
    itemset_dict = {frozenset(k): v for k, v in freq_itemsets}
    for itemset in itemset_dict:
        if len(itemset) >= 2:
            for i in range(1, len(itemset)):
                for antecedent in combinations(itemset, i):
                    antecedent = frozenset(antecedent)
                    consequent = itemset - antecedent
                    if consequent:
                        confidence = itemset_dict[itemset] / itemset_dict.get(antecedent, 1)
                        if confidence >= min_confidence:
                            rules.append((set(antecedent), set(consequent), confidence))
    return rules
# Run Apriori
frequent_itemsets = get_frequent_itemsets(transactions, min_support=0.1)
rules = generate_rules(frequent_itemsets, min_confidence=0.6)
# Display results
print("Prerna Gyanchandani RNo 6")
print("Frequent Itemsets:")
for item, support in frequent_itemsets:
    print(f"{set(item)}: support = {support:.2f}")
print("\nAssociation Rules:")
for antecedent, consequent, confidence in rules:
    print(f"{antecedent} => {consequent} (confidence = {confidence:.2f})")
Output:
